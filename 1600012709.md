int odd_ones(unsigned x)
{
	int a = (x >> 16) ^ x;
	int b = (a >> 8) ^ a;
	int c = (b >> 4) ^ b;
	int d = (c >> 2) ^ c;
	int e = (d >> 1) ^ d;
	return !(e & 1);
}
