float_bits float_negate(float_bits f)
{
    unsigned sign = f & 0x80000000;
    unsigned exp = f & 0x7f800000;
    unsigned frac = f & 0x7fffff;
    if (exp == 0x7f800000 && frac == 0)
        return f;
    else
    {
        sign ^= 0x80000000;
        f = sign | exp | frac;
    }
        return f;
}
